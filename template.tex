% Custom \cleardoublepage to include "intentionally left blank"
\makeatletter
\renewcommand*{\cleardoublepage}{%
  \clearpage
  \if@twoside
    \ifodd\c@page\else
      \null
      \vspace*{\fill}%
      \begin{center}%
        \textit{This page is intentionally left blank.}%
      \end{center}%
      \vspace{\fill}%
      \thispagestyle{empty}%
      \newpage%
    \fi
  \fi
}
\makeatother

\pretocmd{\section}{%
  \iftoggle{skipclearpage}{%
    \global\togglefalse{skipclearpage}%
  }{%
    \cleardoublepage%
  }%
  \thispagestyle{plain}%
}{}{}
