\makeatletter
\renewcommand*{\cleardoublepage}{%
  \clearpage
  \if@twoside
    \ifodd\c@page\else
      \null
      \vspace*{\fill}
      \begin{center}
        \textit{This page is intentionally left blank.}
      \end{center}
      \vspace{\fill}
      \thispagestyle{empty}
      \newpage
    \fi
  \fi
}
\makeatother

% Before every numbered \section: start on odd page, plain style,
% and switch to arabic numbering on the very first one.
\pretocmd{\section}{%
  \cleardoublepage%
  \iftoggle{firstsection}{%
    \pagenumbering{arabic}%
    \global\togglefalse{firstsection}%
  }{}%
  \thispagestyle{plain}%
}{}{}
